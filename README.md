#include <iostream> - подключение библиотеки

using namespace std;

int max(const int* arr, int size) {

    int maxVal = arr[0];
    for (int i = 1; i < size; i++) {
        if (arr[i] > maxVal) {
            maxVal = arr[i];
        }
    }
    return maxVal;
} 
max - функция для нахождения максимума.

Она принимает:
*arr - указатель на массив.
size - количество элементов в массиве.

Maxval - максимальное значение, в начале присвоим первый элеиент массива.

Далее идёт поиск максимума: перебираем элементы массива от i, пока i меньше размера массива, если текущий элемент от i больше максимального, то присваем максимальному значению значение текущего жлемента

В конце функция возвращает найденный максимум.

int main() {
    
    setlocale(LC_ALL, "RU"); - подключаем русский
    int n;
    cout << "Введите количество чисел: "; - выводим сообщение
    cin >> n; - принимаем знчение количества чисел

    int* arr = new int[n]; - выделяем динамически память для массива

    cout << "Введите числа:\n";
    for (int i = 0; i < n; i++) { - используя цикл, принимаем значения
        cin >> arr[i];
    }

    int maximum = max(arr, n); - вводим переменную ддля максимума и присваем ему значение, которое выдаст функция max
    cout << "Максимум: " << maximum << endl; - выводим максимальное значение

    delete[] arr; - освобождаем память
    return 0;
}
